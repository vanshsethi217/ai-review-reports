def get_repo_files(owner, repo, path=""):
    url = "https://api.github.com/repos/" + owner + "/" + repo + "/contents/" + path
    headers = {
        "Authorization": "Bearer " + GITHUB_TOKEN,
        "Accept": "application/vnd.github+json"
    }

    response = requests.get(url, headers=headers)
    response.raise_for_status()
    items = response.json()

    skip_folders = [".git", "node_modules", "venv", "__pycache__", ".github"]
    skip_extensions = [".png", ".jpg", ".jpeg", ".gif", ".pdf", ".zip", ".lock", ".svg", ".ico"]

    file_paths = []

    for item in items:
        if item["type"] == "dir":
            if item["name"] in skip_folders:
                continue
            file_paths.extend(get_repo_files(owner, repo, item["path"]))
        elif item["type"] == "file":
            if any(item["name"].endswith(ext) for ext in skip_extensions):
                continue
            if item["name"].startswith("scan_") and item["name"].endswith(".html"):
                continue
            file_paths.append(item["path"])

    return file_paths
